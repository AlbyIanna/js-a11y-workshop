# prefers-reduced-motion

<Code title="CSS" side="left">

```css
@media (prefers-reduced-motion: reduce) {
    .animation {
        animation: none;
        transition: none;
    }
}
```

</Code>

<Code title="JavaScript" side="right">

```js
var motionQuery = matchMedia('(prefers-reduced-motion)');
function handleReduceMotionChanged() {
  if (motionQuery.matches) {
    /* adjust 'transition' or 'animation' properties */
  } else { 
    /* standard motion */
  }
}
motionQuery.addListener(handleReduceMotionChanged);
handleReduceMotionChanged(); // trigger once on load
```

</Code>

[MDN docs guide](https://developer.mozilla.org/en-US/docs/Web/CSS/@media/prefers-reduced-motion)

# Puppeteer Accessibility tree example

https://pptr.dev/#?product=Puppeteer&version=v1.14.0&show=api-accessibilitysnapshotoptions

```js
const snapshot = await page.accessibility.snapshot();
const node = findFocusedNode(snapshot);
console.log(node && node.name);

function findFocusedNode(node) {
  if (node.focused)
    return node;
  for (const child of node.children || []) {
    const foundNode = findFocusedNode(child);
    return foundNode;
  }
  return null;
}
```
